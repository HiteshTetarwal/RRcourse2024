convert_to_sqft <- function(area) {
  if (grepl("sqft", area)) {
    as.numeric(sub(" sqft", "", area))
  } else if (grepl("sqm", area)) {
    as.numeric(sub(" sqm", "", area)) * 10.7639
  } else {
    NA
  }

# Define a threshold for the minimum number of records per class
